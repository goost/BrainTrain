170  bool FPSCameraManipulator::performEyeMovement() {

172  OSG_DEBUG << "Intensity= " << intensity << std::endl;
173 
174  /*
175  * COLLISION DETECTION AND CLAMPING INSPIRED BY *

177  * AND DRIVEMANIPULATOR *
178  */

180  //newEye tmp 'cos we do not know, if the new position is valid
181  auto newEye = _eye;
182  //up, right, forward dependent on the view
183  auto up = _rotation * osg::Vec3d(0.0, 1.0, 0.0);
184  auto forward = _rotation * osg::Vec3d(0.0, 0.0, -1.0);
185  auto right = _rotation* osg::Vec3d(1.0, 0.0, 0.0);
186  osg::Vec3d movement;
187 
188  //where to move
189  if (_rightMovement)
190  movement += right;
191  if (_leftMovement)
192  movement += -right;
193  if (_forwardMovement)
194  movement += forward * 1.2;
195  if (_backwardMovement)

197  if (_upMovement)
198  movement += up;
199  if (_downMovement)
200  movement += -up;

202  //_delta_frame_time -> not FPS depended, _frameFactor->'cos _deltaTime is so small
203  movement *= _movementSpeed * intensity * _delta_frame_time * _frameFactor;
204 
205  if (_flightMode) {

207  _eye = newEye;
208  }//if(_flightMode)
209  else {
